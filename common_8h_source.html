20bool DecodeWeather(WiFiClient& json, String Type) {
21 log_d("Creating JSON object");
22 // allocate the JsonDocument

24

26 DeserializationError error = deserializeJson(doc, json);

31 }
32 // convert it to a JsonObject
33 JsonObject root = doc.as<JsonObject>();
34 log_d("Decoding %s data", Type.c_str());
35 if (Type == "weather") {
36 WxConditions[0].lon = root["coord"]["lon"].as<float>(); log_d(" Lon: %f", WxConditions[0].lon);
37 WxConditions[0].lat = root["coord"]["lat"].as<float>(); log_d(" Lat: %f", WxConditions[0].lat);
38 WxConditions[0].Main0 = root["weather"][0]["main"].as<const char*>(); log_d("Main: %s", WxConditions[0].Main0);
39 WxConditions[0].Forecast0 = root["weather"][0]["description"].as<const char*>(); log_d("For0: %s", WxConditions[0].Forecast0);
40 WxConditions[0].Forecast1 = root["weather"][1]["description"].as<const char*>(); log_d("For1: %s", WxConditions[0].Forecast1);

43 WxConditions[0].Temperature = root["main"]["temp"].as<float>(); log_d("Temp: %f", WxConditions[0].Temperature);

46 WxConditions[0].Low = root["main"]["temp_min"].as<float>(); log_d("TLow: %f", WxConditions[0].Low);
47 WxConditions[0].High = root["main"]["temp_max"].as<float>(); log_d("THig: %f", WxConditions[0].High);
48 WxConditions[0].Windspeed = root["wind"]["speed"].as<float>(); log_d("WSpd: %f", WxConditions[0].Windspeed);
49 WxConditions[0].Winddir = root["wind"]["deg"].as<float>(); log_d("WDir: %f", WxConditions[0].Winddir);
50 WxConditions[0].Cloudcover = root["clouds"]["all"].as<int>(); log_d("CCov: %d", WxConditions[0].Cloudcover); // in % of cloud cover
51 WxConditions[0].Visibility = root["visibility"].as<int>(); log_d("Visi: %d", WxConditions[0].Visibility); // in metres
52 WxConditions[0].Rainfall = root["rain"]["1h"].as<float>(); log_d("Rain: %f", WxConditions[0].Rainfall);

58 }
59 if (Type == "forecast") {
60 //log_d(json);

62 JsonArray list = root["list"];
63 for (byte r = 0; r < max_readings; r++) {

66 WxForecast[r].Temperature = list[r]["main"]["temp"].as<float>(); log_d("Temp: %f", WxForecast[r].Temperature);

69 WxForecast[r].Pressure = list[r]["main"]["pressure"].as<float>(); log_d("Pres: %f", WxForecast[r].Pressure);
70 WxForecast[r].Humidity = list[r]["main"]["humidity"].as<float>(); log_d("Humi: %f", WxForecast[r].Humidity);

72 WxForecast[r].Forecast1 = list[r]["weather"][1]["main"].as<const char*>(); log_d("For1: %s", WxForecast[r].Forecast1);

74 WxForecast[r].Icon = list[r]["weather"][0]["icon"].as<const char*>(); log_d("Icon: %s", WxForecast[r].Icon);
75 WxForecast[r].Description = list[r]["weather"][0]["description"].as<const char*>(); log_d("Desc: %s", WxForecast[r].Description);
76 WxForecast[r].Cloudcover = list[r]["clouds"]["all"].as<int>(); log_d("CCov: %d", WxForecast[r].Cloudcover); // in % of cloud cover
77 WxForecast[r].Windspeed = list[r]["wind"]["speed"].as<float>(); log_d("WSpd: %f", WxForecast[r].Windspeed);
78 WxForecast[r].Winddir = list[r]["wind"]["deg"].as<float>(); log_d("WDir: %f", WxForecast[r].Winddir);
79 WxForecast[r].Rainfall = list[r]["rain"]["3h"].as<float>(); log_d("Rain: %f", WxForecast[r].Rainfall);
80 WxForecast[r].Snowfall = list[r]["snow"]["3h"].as<float>(); log_d("Snow: %f", WxForecast[r].Snowfall);
81 WxForecast[r].Pop = list[r]["pop"].as<float>(); log_d("Pop: %f", WxForecast[r].Pop);
82 WxForecast[r].Period = list[r]["dt_txt"].as<const char*>(); log_d("Peri: %s", WxForecast[r].Period);
83 }
84 //------------------------------------------
85 float pressure_trend = WxForecast[2].Pressure - WxForecast[0].Pressure; // Measure pressure slope between ~now and later
86 pressure_trend = ((int)(pressure_trend * 10)) / 10.0; // Remove any small variations less than 0.1
87 WxConditions[0].Trend = "0";
88 if (pressure_trend > 0) WxConditions[0].Trend = "+";
89 if (pressure_trend < 0) WxConditions[0].Trend = "-";
90 if (pressure_trend == 0) WxConditions[0].Trend = "0";
91
92 if (Units == "I") Convert_Readings_to_Imperial();
93 }
94 return true;
95}

97String ConvertUnixTime(int unix_time) {
98 // Returns either '21:12 ' or ' 09:12pm' depending on Units mode
99 time_t tm = unix_time;
100 struct tm *now_tm = gmtime(&tm);
101 char output[40];
102 if (Units == "M") {
103 strftime(output, sizeof(output), "%H:%M %d/%m/%y", now_tm);
104 }
105 else {
106 strftime(output, sizeof(output), "%I:%M%P %m/%d/%y", now_tm);
107 }
108 return output;
109}
